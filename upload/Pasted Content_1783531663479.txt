function ipfApp() {
    var S = window.__IPF || {};
    return {
        mobileMenu: false,
        servicesOpen: false,
        scrolled: false,
        sidebar: true,
        currentTab: S.currentTab || 'dashboard',
        authModal: false,
        authMode: 'login',
        chatOpen: false,
        paymentModal: false,
        paymentPlan: { kicker: '', name: '', amount: '', decimal: '', period: '', plan_slug: '', precio_cents: 0 },
        paymentStep: 1,
        paymentResult: { plan_name: '', fecha_inicio: '', fecha_vencimiento: '' },
        _personalData: {},
        culqiConfigured: S.culqiConfigured || false,
        culqiPublicKey: S.culqiPublicKey || '',
        isLoggedIn: S.isLoggedIn || false,
        userName: S.userName || '',
        userEmail: S.userEmail || '',
        userProfile: S.userProfile || {},

        async handleAuth(e) {
            var formData = new FormData(e.target);
            var data = Object.fromEntries(formData.entries());
            var endpoint = this.authMode === 'login' ? 'auth-login.php' : 'auth-registro.php';
            try {
                var response = await fetch(endpoint, {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify(data)
                });
                var result = await response.json();
                if (result.success) {
                    this.isLoggedIn = true;
                    this.authModal = false;
                    if (this.paymentPlan.kicker) {
                        this.paymentModal = true;
                    } else {
                        window.location.href = 'dashboard.php';
                    }
                } else {
                    alert(result.message);
                }
            } catch (error) {
                alert('Error de conexión');
            }
        },

        initPayment(plan) {
            var kicker = plan.kicker || '';
            var name = plan.name || plan.nombre || '';
            var amount = plan.amount || plan.precio_monto || '';
            var decimal = plan.decimal || plan.precio_decimal || '';
            var periodo = plan.periodo || '';
            var period = plan.period || '';
            if (!period && periodo) {
                period = periodo === 'mensual' ? 'por mes' : 'por año';
            }
            var nivel = kicker.toLowerCase().includes('premium') ? 'premium' : 'basico';
            var tipo = periodo || (name.toLowerCase().includes('anual') ? 'anual' : 'mensual');
            var total = parseFloat(amount + decimal) || 0;

            this.paymentPlan = {
                kicker: kicker, name: name, amount: amount, decimal: decimal, period: period,
                plan_slug: nivel + '_' + tipo,
                precio_cents: Math.round(total * 100),
                culqi_plan_id: plan.culqi_plan_id || ''
            };

            if (!this.isLoggedIn) {
                this.authMode = 'register';
                this.authModal = true;
                return;
            }
            this.paymentStep = 1;
            this.paymentModal = true;
            this.$nextTick(function() { lucide.createIcons(); });
        },

        openCulqiCheckout() {
            if (!this.culqiConfigured) {
                alert('La pasarela de pago no está configurada.');
                return;
            }

            var planId = this.paymentPlan.culqi_plan_id;
            if (!planId) {
                alert('Este plan no está configurado para pagos. Contacte al administrador.');
                return;
            }

            var form = document.getElementById('payment-form');
            var fd = new FormData(form);
            this._personalData = Object.fromEntries(fd.entries());

            Culqi.publicKey = this.culqiPublicKey;
            Culqi.settings({
                title: 'Instituto Pulso Fiscal',
                currency: 'PEN',
                amount: this.paymentPlan.precio_cents,
                description: this.paymentPlan.kicker + ' ' + this.paymentPlan.name
            });
            Culqi.options({
                lang: 'es',
                installments: false,
                paymentMethods: {
                    tarjeta: true,
                    yape: true,
                    bancaMovil: false,
                    agente: false,
                    billetera: false,
                    cuotealo: false
                }
            });
            Culqi.open();
        },

        async handleCulqiToken(tokenId) {
            this.paymentStep = 2;
            try {
                var response = await fetch('procesar-pago.php', {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify({
                        token_id: tokenId,
                        plan_slug: this.paymentPlan.plan_slug,
                        personal: this._personalData
                    })
                });
                var result = await response.json();
                if (result.success) {
                    this.paymentStep = 3;
                    this.paymentResult = result.data;
                    this.$nextTick(function() { lucide.createIcons(); });
                } else {
                    alert('Error: ' + result.message);
                    this.paymentStep = 1;
                }
            } catch (e) {
                console.error(e);
                alert('Error de conexión con el servidor.');
                this.paymentStep = 1;
            }
        }
    };
}
</script>
