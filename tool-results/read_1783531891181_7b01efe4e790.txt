   152→   152→        mobileMenu: false,
   153→   153→        servicesOpen: false,
   154→   154→        scrolled: false,
   155→   155→        sidebar: true,
   156→   156→        currentTab: S.currentTab || 'dashboard',
   157→   157→        authModal: false,
   158→   158→        authMode: 'login',
   159→   159→        chatOpen: false,
   160→   160→        paymentModal: false,
   161→   161→        paymentPlan: { kicker: '', name: '', amount: '', decimal: '', period: '', plan_slug: '', precio_cents: 0 },
   162→   162→        paymentStep: 1,
   163→   163→        paymentResult: { plan_name: '', fecha_inicio: '', fecha_vencimiento: '' },
   164→   164→        _personalData: {},
   165→   165→        culqiConfigured: S.culqiConfigured || false,
   166→   166→        culqiPublicKey: S.culqiPublicKey || '',
   167→   167→        isLoggedIn: S.isLoggedIn || false,

   169→   169→        userEmail: S.userEmail || '',
   170→   170→        userProfile: S.userProfile || {},
   171→   171→
   172→   172→        async handleAuth(e) {
   173→   173→            var formData = new FormData(e.target);
   174→   174→            var data = Object.fromEntries(formData.entries());
   175→   175→            var endpoint = this.authMode === 'login' ? 'auth-login.php' : 'auth-registro.php';
   176→   176→            try {
   177→   177→                var response = await fetch(endpoint, {
   178→   178→                    method: 'POST',
   179→   179→                    headers: { 'Content-Type': 'application/json' },
   180→   180→                    body: JSON.stringify(data)
   181→   181→                });
   182→   182→                var result = await response.json();
   183→   183→                if (result.success) {
   184→   184→                    this.isLoggedIn = true;
   185→   185→                    this.authModal = false;
   186→   186→                    if (this.paymentPlan.kicker) {
   187→   187→                        this.paymentModal = true;
   188→   188→                    } else {
   189→   189→                        window.location.href = 'dashboard.php';
   190→   190→                    }
   191→   191→                } else {
   192→   192→                    alert(result.message);
   193→   193→                }
   194→   194→            } catch (error) {
   195→   195→                alert('Error de conexión');
   196→   196→            }
   197→   197→        },
   198→   198→
   199→   199→        initPayment(plan) {
   200→   200→            var kicker = plan.kicker || '';
   201→   201→            var name = plan.name || plan.nombre || '';
   202→   202→            var amount = plan.amount || plan.precio_monto || '';
   203→   203→            var decimal = plan.decimal || plan.precio_decimal || '';
   204→   204→            var periodo = plan.periodo || '';
   205→   205→            var period = plan.period || '';
   206→   206→            if (!period && periodo) {
   207→   207→                period = periodo === 'mensual' ? 'por mes' : 'por año';
   208→   208→            }
   209→   209→            var nivel = kicker.toLowerCase().includes('premium') ? 'premium' : 'basico';
   210→   210→            var tipo = periodo || (name.toLowerCase().includes('anual') ? 'anual' : 'mensual');
   211→   211→            var total = parseFloat(amount + decimal) || 0;
   212→   212→
   213→   213→            this.paymentPlan = {
   214→   214→                kicker: kicker, name: name, amount: amount, decimal: decimal, period: period,
   215→   215→                plan_slug: nivel + '_' + tipo,
   216→   216→                precio_cents: Math.round(total * 100),
   217→   217→                culqi_plan_id: plan.culqi_plan_id || ''
   218→   218→            };
   219→   219→
   220→   220→            if (!this.isLoggedIn) {
   221→   221→                this.authMode = 'register';
   222→   222→                this.authModal = true;
   223→   223→                return;
   224→   224→            }
   225→   225→            this.paymentStep = 1;
   226→   226→            this.paymentModal = true;
   227→   227→            this.$nextTick(function() { lucide.createIcons(); });
   228→   228→        },
   229→   229→
   230→   230→        openCulqiCheckout() {
   231→   231→            if (!this.culqiConfigured) {
   232→   232→                alert('La pasarela de pago no está configurada.');
   233→   233→                return;
   234→   234→            }
   235→   235→
   236→   236→            var planId = this.paymentPlan.culqi_plan_id;
   237→   237→            if (!planId) {
   238→   238→                alert('Este plan no está configurado para pagos. Contacte al administrador.');
   239→   239→                return;
   240→   240→            }
   241→   241→
   242→   242→            var form = document.getElementById('payment-form');
   243→   243→            var fd = new FormData(form);
   244→   244→            this._personalData = Object.fromEntries(fd.entries());
   245→   245→
   246→   246→            Culqi.publicKey = this.culqiPublicKey;
   247→   247→            Culqi.settings({
   248→   248→                title: 'Instituto Pulso Fiscal',
   249→   249→                currency: 'PEN',
   250→   250→                amount: this.paymentPlan.precio_cents,
   251→   251→                description: this.paymentPlan.kicker + ' ' + this.paymentPlan.name
   252→   252→            });
   253→   253→            Culqi.options({
   254→   254→                lang: 'es',
   255→   255→                installments: false,
   256→   256→                paymentMethods: {
   257→   257→                    tarjeta: true,
   258→   258→                    yape: true,
   259→   259→                    bancaMovil: false,
   260→   260→                    agente: false,
   261→   261→                    billetera: false,
   262→   262→                    cuotealo: false
   263→   263→                }
   264→   264→            });
   265→   265→            Culqi.open();
   266→   266→        },
   267→   267→
   268→   268→        async handleCulqiToken(tokenId) {
   269→   269→            this.paymentStep = 2;
   270→   270→            try {
   271→   271→                var response = await fetch('procesar-pago.php', {
   272→   272→                    method: 'POST',
   273→   273→                    headers: { 'Content-Type': 'application/json' },
   274→   274→                    body: JSON.stringify({
   275→   275→                        token_id: tokenId,
   276→   276→                        plan_slug: this.paymentPlan.plan_slug,
   277→   277→                        personal: this._personalData
   278→   278→                    })
   279→   279→                });
   280→   280→                var result = await response.json();
   281→   281→                if (result.success) {
   282→   282→                    this.paymentStep = 3;
   283→   283→                    this.paymentResult = result.data;
   284→   284→                    this.$nextTick(function() { lucide.createIcons(); });
   285→   285→                } else {
   286→   286→                    alert('Error: ' + result.message);
   287→   287→                    this.paymentStep = 1;
   288→   288→                }
   289→   289→            } catch (e) {
   290→   290→                console.error(e);
   291→   291→                alert('Error de conexión con el servidor.');
   292→   292→                this.paymentStep = 1;
   293→   293→            }
   294→   294→        }
   295→   295→    };
